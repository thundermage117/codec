cmake_minimum_required(VERSION 3.16)

project(codec_core VERSION 1.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED True)

# Find OpenCV
find_package(OpenCV REQUIRED)

include_directories(
    inc
    ${OpenCV_INCLUDE_DIRS}
)

add_executable(codec_core
    src/main.cpp
    src/image.cpp
)

target_link_libraries(codec_core
    ${OpenCV_LIBS}
)
